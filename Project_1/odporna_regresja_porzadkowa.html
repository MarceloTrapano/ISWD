def add_monotonicity_constraints(prob: LpProblem, breakpoints: List[float]) -> Tuple[LpProblem, List[List[LpVariable]], List[LpVariable], List[LpVariable]]:
    u_a = [] # zmienne u_i(a_i), gdzie a_i to najgorsza warto kryterium i
    u_b = [] # zmienne u_i(b_i), gdzie b_i to najlepsza warto kryterium i
    u_breakpoints = [] # macierz ixj zmienne u_i(x_i^j), gdzie x_i^j to j-ty punkt przeamania dla kryterium i

    # ograniczenia monotonicznoci
    for i, dir in enumerate(criterion_directions):
        u_bp = []
        for j, bp in enumerate(breakpoints[i]):
            u_bp.append(LpVariable(name=f"u_{i+1}({bp})", lowBound=0, upBound=1)) # zmienne u_i(x_i^j) wyraajce uyteczno dla kadego kryterium i punktu przeamania

        # dodanie ogranicze monotonicznoci
        if dir == "gain":
            for j in range(len(u_bp) - 1):
                # im wiksza warto na kryterium tym wiksza uyteczno
                prob += u_bp[j] <= u_bp[j + 1], f"Monotonicity_{criterions[i]}_{j}"
            u_a.append(u_bp[0])
            u_b.append(u_bp[-1])
        else:
            for j in range(len(u_bp) - 1):
                # im wiksza warto na kryterium tym mniejsza uyteczno
                prob += u_bp[j] >= u_bp[j + 1], f"Monotonicity_{criterions[i]}_{j}"
            u_a.append(u_bp[-1])
            u_b.append(u_bp[0])
        u_breakpoints.append(u_bp)
    return prob, u_breakpoints, u_a, u_b

def add_normalization_constraints(prob: LpProblem, u_a: List[LpVariable], u_b: List[LpVariable]) -> LpProblem:
    # suma nalepszych wartoci uytecznoci = 1
    prob += lpSum(u_b) == 1, "Sum_of_u_b=1"
    # najgorsze uytecznoci = 0
    for i, ua in enumerate(u_a):
        prob += ua == 0, f"u_a_{i+1}=0"
    return prob

def calculate_variant_utility(nr: int, u_breakpoints: List[List[LpVariable]], breakpoints: List[float]) -> Tuple[LpAffineExpression, List[LpVariable]]:
    local_utilities = []
    for i, bp in enumerate(breakpoints):
        u_bp = u_breakpoints[i] # zmienne uytecznoci dla punktw przamania (breakpoints) na kryterium i
        diff = diffs[i] / (len(breakpoints[i]) - 1) # rnica wartoci kryterium midzy punktami kolejnymi przeamania
        val_c = data[nr][i + 1] # warto kryterium i dla wariantu nr

        for j in range(len(bp) - 1):
            if val_c < bp[j + 1]:
                break
        # (j, j+1) - numery punktw przeamania midzy ktrymi naley obliczy interpolacj
        start = j
        end = j + 1
        # uyteczno na kryterium to uyteczno w pocztkowym punkcie przeamania + interpolacja liniowa midzy punktami przeamania
        if val_c == bp[j]: # jeli warto kryterium jest rwna punktowi przeamania to uyteczno jest rwna uytecznoci w tym punkcie - nie trzeba interpolowa
            local_utilities.append(u_bp[start])
        else:
            local_utilities.append(u_bp[start] + (val_c - bp[start]) * (u_bp[end] - u_bp[start]) / diff)
    return lpSum(local_utilities), local_utilities # globalna uyteczno wariantu to suma lokalnych uytecznoci

# po to by wygodnie mie dostp do zmiennych globalnych uytecznoci wariantw, dodadkowo lokalne uytecznoci wariantw
def vars_utilities(prob: LpProblem, u_breakpoints: List[List[LpVariable]], breakpoints: List[float]) -> Tuple[LpProblem, List[LpVariable], List[List[LpVariable]]]:
    global_utilities = []
    local_utilities = []
    # dla kadego wariantu dodajemy zmienn globalnej uytecznoci
    # oraz ograniczenie wyraajce j jako sum lokalnych uytecznoci
    for i in range(len(data)):
        global_utilities.append(LpVariable(name=f"U_x{i+1}")) # ograniczenia na przedzia [0, 1] niepotrzebne, bo wymuszone przez inne ograniczenia
        g_util, l_utils = calculate_variant_utility(i, u_breakpoints, breakpoints)
        local_utilities.append(l_utils)
        prob += global_utilities[i] == g_util, f"Global_utility_x{i+1}"
    return prob, global_utilities, local_utilities

UTA

Definicja problemu

Problem jest zdefiniowany nastpujco: $$\max \quad \varepsilon$$ Ograniczenia preferencji: $$U(x_{26})\ge U(x_{15}) + \varepsilon$$ $$U(x_{19})\ge U(x_{12}) + \varepsilon$$ $$U(x_{7})\ge U(x_{24}) + \varepsilon$$ $$U(x_{14})= U(x_{17})$$ $$U(x_{10})= U(x_{8})$$ Ograniczenia monotonicznoci: $$u_{C_1}(1) < u_{C_1}(0.5) < u_{C_1}(0)$$ $$u_{C_2}(0) < u_{C_2}(0.5) < u_{C_2}(1)$$ $$u_{C_3}(0) < u_{C_3}(0.5) < u_{C_3}(1)$$ $$u_{C_4}(1) < u_{C_4}(0.5) < u_{C_4}(0)$$ Ograniczenia normalizacji: $$u_{C_1}(0) + u_{C_2}(1) + u_{C_3}(1) + u_{C_4}(0) = 1$$ $$u_{C_1}(1) = 0$$ $$u_{C_2}(0) = 0$$ $$u_{C_3}(0) = 0$$ $$u_{C_4}(1) = 0$$ Oprcz wszystkich wymaganych w UTA ogranicze dodano ograniczenie na maksymaln uyteczno kadego kryterium, ktra powinna wynosi co najmniej 0.1 (tre zadania). Czyli $$ \sum_{i=1}^{n}u_i\left(\beta_i\right) \ge 0.1$$ gdzie $\beta_i$ to naljepsza warto na kryterium $i$

$\text{UTA}^{\text{GMS}}$

W tej metodzie punkty przeamania stanowi wartoci kryteriw wariantw. Przy jej pomocy mona znajdywa relacj moliwej preferencji oraz relacj koniecznej preferencji poprzez dodawanie do problemu nastpujcych ogranicze:

  • do sprawdzenia relacji koniecznej preferencji wariantu $a$ nad $b$ sprawdzamy czy dodanie warunku $$U(b) \ge U(a) + \varepsilon$$ generuje sprzeczno. Wtedy jestemy w stanie stwierdzi, e nie istnieje funkcja dla ktrej wariant $b$ jest cile prefereowany nad $a$, wic relacja przeciwna $a$ preferowany nad $b$ zawsze zachodzi.
  • do sprawdzenia relacji moliwej preferencji wariantu $a$ nad $b$ sprawdzamy czy dodanie warunku $$U(a) \ge U(b)$$ zachodzi. Wtedy ta relacja jest moliwa dla co najmniej jednej funkcji.

            nr_no_transitive.remove((a, b))

for a, b in nr_no_transitive:
    m_start[a][b] = 1 # a>=Nb

groups = [] # grupy - wierzchoki (node-y) w grafie
finished = [] # tablica do oznaczania zakoczonych wariantw

# szuka kolejno grup - niezdominowane warianty dla aktualnej iteracji, ktre maj takie same relacje w m_start
# kolejne grupy bd nie wyej w hierarchii ni poprzednie - dlatego w komrce niej taki warunek sample_variants_to_group[a] < i
while len(finished) < num_variants:
    sum_col = np.sum(m, axis=0) # liczba wariantw ktre domunuj wariant
    non_dominated = np.where(sum_col == 0)[0] # warianty niedominowane
    non_dominated = np.delete(non_dominated, np.where(np.isin(non_dominated, finished))[0]) # usunicie zakoczonych wariantw
    if np.sum(non_dominated) == 0: # jeli nie ma niedominowanych to koniec
        break
    while non_dominated.size > 0:
        # dla kadego wariantu niedominowanego
        curr_group = [non_dominated[0]]
        # pierwszy wariant z grupy
        j = non_dominated[0]
        for i in range(1, len(non_dominated)):
            # potencjalny wariant do grupy
            k = non_dominated[i]
            # jeli k ma takie same relacje w m_start co j to dodaj do grupy
            if np.array_equal(m_start[k], m_start[j]) and np.array_equal(m_start[:, k], m_start[:, j]):

        for i in curr_group:

            m[:,i] = 0

Reprezentatywna funkcja uytecznoci

Do znalezienia reprezentatywnej funkcji uytecznoci do metody UTA GMS naley doda do problemu relacje konieczne i moliwe:

  • jeli zachodzi relacja koniecznej preferencji wariantu $a$ nad $b$ dodajemy: $$U(a) \ge U(b) + \varepsilon$$
  • jeli nie zachodzi relacja konieczna w adn stron: $$U(a) - U(b)\le \delta$$ $$U(b) - U(a)\le \delta$$ gdzie $\delta$ jest minimalizowana
  • do zbioru ogranicze naley take doda $\varepsilon = \varepsilon^*$
